#include <iostream>
#include <vector>
#include <iomanip>
#include <algorithm>
using namespace std;

struct Buku {
    int idBuku; 
    string judulBuku; 
    string penulis; 
    bool isTersedia;
};

void tampilkanBukuTerurut(vector<Buku> listBuku) { 
    if (listBuku.empty()) {
        cout << "\nRak buku masih kosong.\n"; return;
    }
    
    // [NOTE ALGORITMA: Bubble Sort (Pengurutan Gelembung)]
    int n = listBuku.size();
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (listBuku[j].judulBuku > listBuku[j+1].judulBuku) {
                swap(listBuku[j], listBuku[j+1]); 
            }
        }
    }
    
    // [NOTE ALGORITMA: Formating dengan iomanip]
    cout << "\n====================================================================\n";
    cout << "                 DAFTAR KOLEKSI BUKU (TERURUT A-Z)                  \n";
    cout << "====================================================================\n";
    cout << left << setw(10) << "ID" << setw(30) << "JUDUL BUKU" << setw(20) << "PENULIS" << "STATUS\n";
    cout << "--------------------------------------------------------------------\n";
    for (const auto& b : listBuku) {
        cout << left << setw(10) << b.idBuku << setw(30) << b.judulBuku << setw(20) << b.penulis 
             << (b.isTersedia ? "Tersedia" : "Dipinjam") << "\n";
    }
    cout << "====================================================================\n";
}

void cariBuku(const vector<Buku>& listBuku) {
    int idCari;
    cout << "\n=========================================\n";
    cout << "           PENCARIAN DATA BUKU           \n";
    cout << "=========================================\n";
    cout << "Masukkan ID Buku yang dicari: ";
    cin >> idCari;

    // [NOTE ALGORITMA: Linear Search / Pencarian Lurus]
    for (const auto& b : listBuku) {
        if (b.idBuku == idCari) {
            cout << "\n-> DITEMUKAN: " << b.judulBuku << " (" << (b.isTersedia ? "Tersedia" : "Dipinjam") << ")\n";
            return;
        }
    }
    cout << "Validasi: Buku dengan ID " << idCari << " tidak ditemukan!\n";
}

int main() {
    vector<Buku> tesData = {
        {102, "Zaman Batu", "Andi", true}, 
        {101, "Algoritma", "Budi", true}
    };
    cout << "--- TESTING FITUR ARIYA ---\n";
    tampilkanBukuTerurut(tesData);
    return 0;
}
